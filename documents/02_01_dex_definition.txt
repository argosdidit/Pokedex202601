CREATE TABLE POKÉDEX1(
POKEID INTEGER PRIMARY KEY,
NO VARCHAR(5),
NAMAE VARCHAR(20),
NAME VARCHAR(20),
SUGATA VARCHAR(20),
FORM VARCHAR(40),
BUNRUI VARCHAR(20),
CLASSIFICATION VARCHAR(25),

TYPE1 INTEGER,
TYPE2 INTEGER,
ABILITY1 INTEGER,
ABILITY2 INTEGER,
HIDDEN_ABILITY INTEGER,
GENDER INTEGER,
EGG_GROUP1 INTEGER,
EGG_GROUP2 INTEGER,
REGION INTEGER,
GENERATION INTEGER,
AUTONUM INTEGER AUTO_INCREMENT,

FOREIGN KEY (TYPE1) REFERENCES TYPE(TYPEID),
FOREIGN KEY (TYPE2) REFERENCES TYPE(TYPEID),
FOREIGN KEY (ABILITY1) REFERENCES ABILITY(ABILITYID),
FOREIGN KEY (ABILITY2) REFERENCES ABILITY(ABILITYID),
FOREIGN KEY (HIDDEN_ABILITY) REFERENCES ABILITY(ABILITYID),
FOREIGN KEY (GENDER) REFERENCES GENDER(GENDERID),
FOREIGN KEY (EGG_GROUP1) REFERENCES EGG_GROUP(EGG_GROUPID),
FOREIGN KEY (EGG_GROUP2) REFERENCES EGG_GROUP(EGG_GROUPID),
FOREIGN KEY (REGION) REFERENCES REGION(REGIONID),
FOREIGN KEY (GENERATION) REFERENCES GENERATION(GENERATIONID),
UNIQUE KEY(AUTONUM)
);

SP_DEF INTEGER,

SUM INTEGER
